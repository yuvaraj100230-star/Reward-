Student Records Management

Program Overview

This program manages student records using structures. It allows users to input student information, calculate total marks and averages, and then sorts students based on their total marks in descending order.

Program Code

#include <iostream>
using namespace std;

struct Student {
string name;
int rno, m1, m2, m3, total;
float avg;
};

int main() {
int n;
cout << "Enter number of students: ";
cin >> n;

Student s[n];  

// Input  
for(int i = 0; i < n; i++) {  
    cout << "\nName: "; cin >> s[i].name;  
    cout << "Roll No: "; cin >> s[i].rno;  
    cout << "Marks 1 2 3: "; cin >> s[i].m1 >> s[i].m2 >> s[i].m3;  
    s[i].total = s[i].m1 + s[i].m2 + s[i].m3;  
    s[i].avg = s[i].total / 3.0;  
}  

// Sort by total (descending)  
for(int i = 0; i < n-1; i++) {  
    for(int j = i+1; j < n; j++) {  
        if(s[i].total < s[j].total) {  
            Student t = s[i];  
            s[i] = s[j];  
            s[j] = t;  
        }  
    }  
}  

// Display  
cout << "\nNAME\tROLL\tM1\tM2\tM3\tTOTAL\tAVG\n";  
for(int i = 0; i < n; i++)  
    cout << s[i].name << "\t" << s[i].rno << "\t"  
         << s[i].m1 << "\t" << s[i].m2 << "\t" << s[i].m3  
         << "\t" << s[i].total << "\t" << s[i].avg << "\n";  

return 0;
}

1. Structure Definition

The program defines a Student structure to store student information:

    int rno, m1, m2, m3, total;

The program takes input for multiple students and calculates their total marks and averages:

    cout << "Roll No: "; cin >> s[i].rno;

    s[i].total = s[i].m1 + s[i].m2 + s[i].m3;

Polynomial Operations

Program Overview

This program performs addition and subtraction operations on cubic polynomials. It takes coefficients for two cubic polynomials as input and allows the user to perform operations through a menu-driven interface.

Program Code

#include <iostream>
using namespace std;

int main() {
    int choice;

    // Coefficients for First Polynomial a1x^3 + b1x^2 + c1x + d1
    int a1, b1, c1, d1;

    // Coefficients for Second Polynomial a2x^3 + b2x^2 + c2x + d2
    int a2, b2, c2, d2;

    cout << "Enter the coefficients for FIRST Polynomial:\n";
    cout << "a1 = "; cin >> a1;
    cout << "b1 = "; cin >> b1;
    cout << "c1 = "; cin >> c1;
    cout << "d1 = "; cin >> d1;

    cout << "\nEnter the coefficients for SECOND Polynomial :\n";
    cout << "a2 = "; cin >> a2;
    cout << "b2 = "; cin >> b2;
    cout << "c2 = "; cin >> c2;
    cout << "d2 = "; cin >> d2;

    while (true) {
        cout << "\n=============================\n";
        cout << "  Polynomial Operations Menu\n";
        cout << "=============================\n";
        cout << "1. Addition of Polynomials\n";
        cout << "2. Subtraction of Polynomials\n";
        cout << "3. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice;

        if (choice == 1) {
            // ADDITION
            int A = a1 + a2;
            int B = b1 + b2;
            int C = c1 + c2;
            int D = d1 + d2;

            cout << "\nAddition Result: "
                 << A << "x^3 + " << B << "x^2 + " << C << "x + " << D << endl;
        }
        else if (choice == 2) {
            // SUBTRACTION
            int A = a1 - a2;
            int B = b1 - b2;
            int C = c1 - c2;
            int D = d1 - d2;

            cout << "\nSubtraction Result: "
                 << A << "x^3 + " << B << "x^2 + " << C << "x + " << D << endl;
        }
        else if (choice == 3) {
            cout << "\nExiting program..." << endl;
            break;
        }
        else {
            cout << "\nInvalid choice! Please try again.\n";
        }
    }

    return 0;
}

1. Variable Declaration

The program declares variables to store the coefficients of two cubic polynomials:

int a1, b1, c1, d1;  // Coefficients for the first polynomial
int a2, b2, c2, d2;  // Coefficients for the second polynomial
