Stack using Array

About Stack

A stack is a linear data structure that follows the Last-In-First-Out (LIFO) principle. This implementation uses an array to create a stack with push, pop, and display operations.

#include <iostream.h> int main() { int stack[100], top = -1; // stack array and top pointer int choice, value; do { cout << "\n1. Push\n2. Pop\n3. Display\n4. Exit\n"; cout << "Enter your choice: "; cin >> choice; if (choice == 1) { cout << "Enter value to push: "; cin >> value; top++; stack[top] = value; cout << "Pushed " << value << endl; } else if (choice == 2) { if (top == -1) cout << "Stack Underflow\n"; else { cout << "Popped: " << stack[top] << endl; top--; } } else if (choice == 3) { if (top == -1) cout << "Stack is Empty\n"; else { cout << "Stack: "; for (int i = top; i >= 0; i--) { cout << stack[i] << " "; } cout << endl; } } } while (choice != 4); return 0; }

Double Linked List

About Double Linked List

A doubly linked list is a linear data structure where each element has pointers to both the next and previous elements. This program demonstrates insertion at the end, deletion from the end, and display operations.

#include <iostream.h> struct Node { int data; Node* prev; Node* next; }; Node* head = NULL; void insertEnd(int value) { Node* newNode = new Node(); newNode->data = value; newNode->next = NULL; if (head == NULL) { newNode->prev = NULL; head = newNode; return; } Node* temp = head; while (temp->next != NULL) temp = temp->next; temp->next = newNode; newNode->prev = temp; } void deleteEnd() { if (head == NULL) { cout << "List is empty!\n"; return; } if (head->next == NULL) { delete head; head = NULL; cout << "Last node deleted.\n"; return; } Node* temp = head; while (temp->next != NULL) temp = temp->next; temp->prev->next = NULL; delete temp; cout << "Node deleted from end.\n"; } void display() { Node* temp = head; cout << "\nDoubly Linked List: "; while (temp != NULL) { cout << temp->data << " <-> "; temp = temp->next; } cout << "NULL\n"; } int main() { int choice, value; do { cout << "\n1. Insert at End\n2. Delete from End\n3. Display\n4. Exit\n"; cout << "Enter your choice: "; cin >> choice; switch (choice) { case 1: cout << "Enter value: "; cin >> value; insertEnd(value); break; case 2: deleteEnd(); break; case 3: display(); break; case 4: cout << "Exiting...\n"; break; default: cout << "Invalid choice!\n"; } } while (choice != 4); return 0; }

Student Marklist

About Student Marklist

This program creates a student marklist using structures and dynamic memory allocation. It allows input of student details including name, roll number, and marks in three subjects, then displays the information in a tabular format.

#include <iostream.h> #include <string.h> #include <conio.h> struct Student { char name[50]; int roll; float m1, m2, m3; }; int main() { clrscr(); int n; cout << "Enter number of students: "; cin >> n; Student* s = new Student[n]; // dynamic array for (int i = 0; i < n; i++) { // Input details cout << "\nEnter details for Student " << i + 1 << endl; cout << "Name: "; cin >> s[i].name; cout << "Roll Number: "; cin >> s[i].roll; cout << "Marks of Subject 1: "; cin >> s[i].m1; cout << "Marks of Subject 2: "; cin >> s[i].m2; cout << "Marks of Subject 3: "; cin >> s[i].m3; } cout << "\n=========== STUDENT MARKLIST ===========\n"; // Display Marklist cout << "Name\tRoll\tSub1\tSub2\tSub3\n"; for (i = 0; i < n; i++) { cout << s[i].name << "\t" << s[i].roll << "\t" << s[i].m1 << "\t" << s[i].m2 << "\t" << s[i].m3 << endl; } getch(); delete[] s; // free memory return 0; }
